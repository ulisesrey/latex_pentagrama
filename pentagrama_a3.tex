\documentclass{article}
\usepackage{tikz}
\usepackage{geometry}
\usepackage{multicol} % For the two-column layout


% --- MARGIN CONFIGURATION ---
\newcommand{\MarginTop}{2cm}
\newcommand{\MarginBottom}{2cm}
\newcommand{\MarginLeft}{2cm}
\newcommand{\MarginRight}{2cm}
\newcommand{\ColumnGap}{4cm} % Space in the middle for the fold/binding

% --- STAFF CONFIGURATION ---
\newcommand{\NumberOfStaves}{30}      
\newcommand{\LineSpacing}{0.15cm}     % Gap between the 5 lines
\newcommand{\GapBetweenStaves}{1.1cm} % Gap between each block of lines
\newcommand{\StaffThickness}{0.6pt}   % Line weight

\geometry{
    a3paper,
    landscape,
    top=\MarginTop,
    bottom=\MarginBottom,
    left=\MarginLeft,
    right=\MarginRight,
    columnsep=\ColumnGap
}

\begin{document}
\pagestyle{empty}

\begin{multicols}{2} % Start the 2-column layout
    \foreach \n in {1,...,\NumberOfStaves}{
        \noindent
        \begin{tikzpicture}
            \foreach \i in {0,1,2,3,4}
            {
                \draw [line width=\StaffThickness] (0, \i*\LineSpacing) -- (\linewidth, \i*\LineSpacing);
            }
        \end{tikzpicture}
        \par
        \vspace{\GapBetweenStaves}
    }
\end{multicols}

\end{document}
